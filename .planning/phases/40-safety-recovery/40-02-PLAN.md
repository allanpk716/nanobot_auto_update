1. `CheckUpdateState(logger *slog.Logger)` -- called early in main() startup:
   ```go
   //go:build windows

   package lifecycle

   import (
       "encoding/json"
       "fmt"
       "log/slog"
       "os"
       "os/exec"
       "syscall"

       "golang.org/x/sys/windows"
   )

   // CheckUpdateState checks for leftover .old backup files from previous updates.
   // Per D-04:
   //   - .update-success exists -> last update succeeded -> clean up .old + delete marker
   //   - .update-success missing but .exe.old exists -> crash during update -> restore from .old
   //   - neither exists -> normal startup, do nothing
   func CheckUpdateState(logger *slog.Logger) {
       exePath, err := os.Executable()
       if err != nil {
           logger.Error("failed to get exe path for update state check", "error", err)
           return
       }

       oldPath := exePath + ".old"
       successPath := exePath + ".update-success"

       // Case 1: .update-success exists -> last update was successful
       if data, err := os.ReadFile(successPath); err == nil {
           // Validate the marker file contains valid JSON
           var marker map[string]string
           if err := json.Unmarshal(data, &marker); err != nil {
               logger.Warn("invalid update-success marker file, treating as corrupt",
                   "path", successPath, "error", err)
               // Corrupt marker: treat as if it doesn't exist -> fall through to Case 2
           } else {
               logger.Info("previous update successful, cleaning up .old backup",
                   "new_version", marker["new_version"],
                   "old_version", marker["old_version"])
               os.Remove(oldPath)     // Clean up .old (may not exist if already cleaned)
               os.Remove(successPath) // Clean up marker
               return
           }
       }

       // Case 2: .old exists but no valid .update-success -> crash during update
       if info, err := os.Stat(oldPath); err == nil && info.Size() > 0 {
           logger.Warn("crash detected during update, restoring from .old backup",
               "old_path", oldPath,
               "exe_path", exePath,
               "old_size", info.Size())

           // Restore using os.Rename (atomic on Windows)
           if err := os.Rename(oldPath, exePath); err != nil {
               logger.Error("failed to restore from .old backup", "error", err)
               return
           }

           logger.Info("restored from .old backup, restarting with recovered version")

           // Self-spawn the restored version
           cmd := exec.Command(exePath, os.Args[1:]...)
           cmd.SysProcAttr = &syscall.SysProcAttr{
               HideWindow:    true,
               CreationFlags: windows.CREATE_NO_WINDOW,
           }
           if err := cmd.Start(); err != nil {
               logger.Error("failed to spawn restored process", "error", err)
               return
           }
           os.Exit(0)
       }

       // Case 3: no .old and no marker -> normal startup
   }
   ```

2. `ListenWithRetry(addr string, logger *slog.Logger) (net.Listener, error)` -- used in Server.Start():
   ```go
   // ListenWithRetry attempts to bind a TCP port with retries.
   // Per D-05: 500ms interval, max 5 retries (2.5s total).
   // Used after self-update restart when old process may still hold the port.
   func ListenWithRetry(addr string, logger *slog.Logger) (net.Listener, error) {
       var lastErr error
       for i := 0; i < 5; i++ {
           listener, err := net.Listen("tcp", addr)
           if err == nil {
               if i > 0 {
                   logger.Info("port bind succeeded after retry",
                       "addr", addr,
                       "attempts", i+1)
               }
               return listener, nil
           }
           lastErr = err
           logger.Warn("port bind failed, retrying",
               "addr", addr,
               "attempt", i+1,
               "error", err)
           time.Sleep(500 * time.Millisecond)
       }
       return nil, fmt.Errorf("failed to bind %s after 5 retries: %w", addr, lastErr)
   }
   ```

   Add required imports: `"net"`, `"time"`, `"encoding/json"`, `"fmt"`.
