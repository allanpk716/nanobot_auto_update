// Self-update module
let authToken = '';
let pollTimer = null;
let pollStartTime = 0;
let isUpdating = false;

// Initialize self-update module (per D-14, D-15)
async function initSelfUpdate() {
    try {
        const resp = await fetch('/api/v1/web-config');
        if (!resp.ok) {
            throw new Error('web-config unavailable');
        }
        const data = await resp.json();
        authToken = data.auth_token;
        // Load current version from check API
        await loadCurrentVersion();
        // Bind button events
        document.getElementById('btn-check-update').addEventListener('click', checkUpdate);
        document.getElementById('btn-start-update').addEventListener('click', startUpdate);
    } catch (e) {
        // D-16: web-config fetch failed (non-localhost)
        console.error('Failed to init self-update:', e);
        const section = document.getElementById('selfupdate-section');
        section.innerHTML = '<p class="selfupdate-warning">请在本地访问以使用自更新功能</p>';
    }
}

// Load current version from check API (per D-03)
async function loadCurrentVersion() {
    try {
        const resp = await fetch('/api/v1/self-update/check', {
            headers: { 'Authorization': 'Bearer ' + authToken }
        });
        if (!resp.ok) {
            throw new Error('check API failed');
        }
        const data = await resp.json();
        document.getElementById('current-version').textContent = data.current_version;
    } catch (e) {
        console.error('Failed to load current version:', e);
        document.getElementById('current-version').textContent = 'N/A';
    }
}

// Check for updates (per D-04, D-05, UI-03)
async function checkUpdate() {
    const btn = document.getElementById('btn-check-update');
    const startBtn = document.getElementById('btn-start-update');
    const resultDiv = document.getElementById('update-result');

    if (isUpdating) return;

    // Disable button during check (per D-13)
    btn.disabled = true;
    btn.textContent = '检测中...';
    startBtn.disabled = true;

    try {
        const resp = await fetch('/api/v1/self-update/check', {
            headers: { 'Authorization': 'Bearer ' + authToken }
        });
        if (!resp.ok) {
            throw new Error('check API returned ' + resp.status);
        }
        const data = await resp.json();

        if (!data.needs_update) {
            // Already up to date
            resultDiv.className = 'update-result visible';
            resultDiv.innerHTML = '<div class="update-info"><span class="info-label">已是最新版本</span><span class="info-value">v' + data.current_version.replace('v', '') + '</span></div>';
            startBtn.disabled = true;
        } else {
            // New version available — render details (per D-04, D-05)
            let releaseNotesHtml = '';
            if (data.release_notes) {
                // Per D-04: truncate to ~4 lines (max-height: 6em), show toggle
                // Use textContent via separate element creation for XSS safety
                const notesDiv = document.createElement('div');
                notesDiv.className = 'release-notes';
                notesDiv.textContent = data.release_notes;

                const toggleBtn = document.createElement('span');
                toggleBtn.className = 'release-notes-toggle';
                toggleBtn.textContent = '展开';
                toggleBtn.addEventListener('click', function() {
                    if (notesDiv.classList.contains('expanded')) {
                        notesDiv.classList.remove('expanded');
                        toggleBtn.textContent = '展开';
                    } else {
                        notesDiv.classList.add('expanded');
                        toggleBtn.textContent = '收起';
                    }
                });

                releaseNotesHtml = '';
            }

            // Format published_at (per D-05: "2026-04-07" format)
            let dateStr = '';
            if (data.published_at) {
                try {
                    const d = new Date(data.published_at);
                    dateStr = d.getFullYear() + '-' +
                        String(d.getMonth() + 1).padStart(2, '0') + '-' +
                        String(d.getDate()).padStart(2, '0');
                } catch (e) {
                    dateStr = data.published_at;
                }
            }

            resultDiv.className = 'update-result visible';
            resultDiv.innerHTML = '';

            const infoDiv = document.createElement('div');
            infoDiv.className = 'update-info';

            const versionRow = document.createElement('div');
            versionRow.style.marginBottom = '4px';
            versionRow.innerHTML = '';
            const versionLabel = document.createElement('span');
            versionLabel.className = 'info-label';
            versionLabel.textContent = '最新版本:';
            const versionValue = document.createElement('span');
            versionValue.className = 'info-value';
            versionValue.textContent = data.latest_version;
            versionRow.appendChild(versionLabel);
            versionRow.appendChild(versionValue);
            infoDiv.appendChild(versionRow);

            if (dateStr) {
                const dateRow = document.createElement('div');
                dateRow.style.marginBottom = '4px';
                const dateLabel = document.createElement('span');
                dateLabel.className = 'info-label';
                dateLabel.textContent = '发布日期:';
                const dateValue = document.createElement('span');
                dateValue.className = 'info-value';
                dateValue.textContent = dateStr;
                dateRow.appendChild(dateLabel);
                dateRow.appendChild(dateValue);
                infoDiv.appendChild(dateRow);
            }

            resultDiv.appendChild(infoDiv);

            // Release notes with expand/collapse (per D-04)
            if (data.release_notes) {
                const notesDiv = document.createElement('div');
                notesDiv.className = 'release-notes';
                notesDiv.textContent = data.release_notes;
                resultDiv.appendChild(notesDiv);

                const toggleBtn = document.createElement('span');
                toggleBtn.className = 'release-notes-toggle';
                toggleBtn.textContent = '展开';
                toggleBtn.addEventListener('click', function() {
                    if (notesDiv.classList.contains('expanded')) {
                        notesDiv.classList.remove('expanded');
                        toggleBtn.textContent = '展开';
                    } else {
                        notesDiv.classList.add('expanded');
                        toggleBtn.textContent = '收起';
                    }
                });
                resultDiv.appendChild(toggleBtn);
            }

            startBtn.disabled = false;
        }
    } catch (e) {
        console.error('Failed to check for updates:', e);
        resultDiv.className = 'update-result visible';
        resultDiv.innerHTML = '<div class="update-error">检测更新失败: ' + e.message + '</div>';
        startBtn.disabled = true;
    } finally {
        btn.disabled = false;
        btn.textContent = '检测更新';
    }
}

// Start update (placeholder — full implementation in Plan 02)
// This function is bound to btn-start-update click event.
// Plan 02 will replace this with full polling logic.
async function startUpdate() {
    const btn = document.getElementById('btn-start-update');
    if (isUpdating || btn.disabled) return;

    try {
        btn.disabled = true;
        document.getElementById('btn-check-update').disabled = true;

        const resp = await fetch('/api/v1/self-update', {
            method: 'POST',
            headers: { 'Authorization': 'Bearer ' + authToken }
        });

        if (resp.status === 409) {
            // Already updating (per Pitfall 4)
            const resultDiv = document.getElementById('update-result');
            resultDiv.className = 'update-result visible';
            resultDiv.innerHTML = '<div class="update-error">更新已在进行中，请稍后再试</div>';
            btn.disabled = false;
            document.getElementById('btn-check-update').disabled = false;
            return;
        }

        if (!resp.ok) {
            throw new Error('update API returned ' + resp.status);
        }

        // Start progress polling (per D-10, D-11)
        isUpdating = true;
        pollStartTime = Date.now();
        startProgressPolling();
    } catch (e) {
        console.error('Failed to start update:', e);
        const resultDiv = document.getElementById('update-result');
        resultDiv.className = 'update-result visible';
        resultDiv.innerHTML = '<div class="update-error">启动更新失败: ' + e.message + '</div>';
        btn.disabled = false;
        document.getElementById('btn-check-update').disabled = false;
    }
}

// Progress polling (per D-10: 500ms, D-11: 60s timeout, D-06: progress bar, D-07: stage text)
function startProgressPolling() {
    const resultDiv = document.getElementById('update-result');
    resultDiv.className = 'update-result visible';
    resultDiv.innerHTML =
        '<div class="progress-container">' +
        '<div class="progress-status" id="progress-status">检查中...</div>' +
        '<div class="progress-bar-track"><div class="progress-bar-fill" id="progress-fill"></div></div>' +
        '<div class="progress-text" id="progress-text"></div>' +
        '</div>';

    pollTimer = setInterval(async function() {
        // D-11: 60 second timeout
        if (Date.now() - pollStartTime > 60000) {
            clearInterval(pollTimer);
            pollTimer = null;
            isUpdating = false;
            resultDiv.innerHTML = '<div class="update-error">更新超时，请检查服务状态</div>';
            document.getElementById('btn-start-update').disabled = false;
            document.getElementById('btn-check-update').disabled = false;
            return;
        }

        try {
            const resp = await fetch('/api/v1/self-update/check', {
                headers: { 'Authorization': 'Bearer ' + authToken }
            });
            if (!resp.ok) {
                // Network error during poll — might be restarting (per Pitfall 3)
                return;
            }
            const data = await resp.json();
            const progress = data.progress;

            // Update progress UI (per D-06, D-07)
            const statusEl = document.getElementById('progress-status');
            const fillEl = document.getElementById('progress-fill');
            const textEl = document.getElementById('progress-text');

            if (!progress || progress.stage === 'idle') {
                if (statusEl) statusEl.textContent = '检查中...';
            } else if (progress.stage === 'checking') {
                if (statusEl) statusEl.textContent = '检查中...';
            } else if (progress.stage === 'downloading') {
                if (statusEl) statusEl.textContent = '下载中 ' + progress.download_percent + '%';
                if (fillEl) fillEl.style.width = progress.download_percent + '%';
                if (textEl) textEl.textContent = progress.download_percent + '%';
            } else if (progress.stage === 'installing') {
                if (statusEl) statusEl.textContent = '安装中...';
                if (fillEl) fillEl.style.width = '100%';
            } else if (progress.stage === 'complete') {
                clearInterval(pollTimer);
                pollTimer = null;
                isUpdating = false;
                // D-08: success feedback
                resultDiv.innerHTML = '<div class="update-success">更新完成，服务即将重启</div>';
                // D-12: reload page after 3 seconds
                setTimeout(function() { location.reload(); }, 3000);
            } else if (progress.stage === 'failed') {
                clearInterval(pollTimer);
                pollTimer = null;
                isUpdating = false;
                // D-09: failure feedback with error info
                const errorMsg = progress.error || '未知错误';
                resultDiv.innerHTML = '<div class="update-error">更新失败: ' + errorMsg + '</div>';
                document.getElementById('btn-start-update').disabled = false;
                document.getElementById('btn-check-update').disabled = false;
            }
        } catch (e) {
            // Network error — service might be restarting (per Pitfall 3)
            // Silently ignore, polling will continue until timeout
            console.log('Poll request failed (service may be restarting):', e.message);
        }
    }, 500);
}
