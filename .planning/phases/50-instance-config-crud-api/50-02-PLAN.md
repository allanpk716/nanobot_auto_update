```go
func TestSaveConfig_WritesInstancesToFile(t *testing.T) {
    tmpDir := t.TempDir()
    configPath := filepath.Join(tmpDir, "config.yaml")

    // Write initial config
    initialYAML := `api:
  port: 9999
  bearer_token: "test-token-12345678901234567890"
instances:
  - name: "existing"
    port: 18790
    start_command: "nanobot gateway"
    startup_timeout: 30s
`
    require.NoError(t, os.WriteFile(configPath, []byte(initialYAML), 0644))

    // Load to initialize viperInstance
    cfg, err := Load(configPath)
    require.NoError(t, err)
    require.Len(t, cfg.Instances, 1)

    // Add a new instance
    cfg.Instances = append(cfg.Instances, InstanceConfig{
        Name:         "new-instance",
        Port:         18791,
        StartCommand: "nanobot gateway --config test",
        StartupTimeout: 30 * time.Second,
    })

    // Save
    err = SaveConfig(cfg)
    require.NoError(t, err)

    // Reload and verify
    newCfg, err := ReloadConfig(cfg)
    require.NoError(t, err)
    require.Len(t, newCfg.Instances, 2)
    assert.Equal(t, "new-instance", newCfg.Instances[1].Name)
    assert.Equal(t, uint32(18791), newCfg.Instances[1].Port)

    // Cleanup
    viperInstance = nil
}
```
