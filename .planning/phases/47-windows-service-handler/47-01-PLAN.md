func AppShutdown(ctx context.Context, c *AppComponents, logger *slog.Logger) {
    if c.NotificationManager != nil { c.NotificationManager.Stop() }
    if c.NetworkMonitor != nil { c.NetworkMonitor.Stop() }
    if c.HealthMonitor != nil { c.HealthMonitor.Stop() }
    if c.CleanupCron != nil {
        c.CleanupCron.Stop()
        logger.Info("Update log cleanup scheduler stopped")
    }
    if c.UpdateLogger != nil {
        if err := c.UpdateLogger.Close(); err != nil {
            logger.Error("Failed to close update logger", "error", err)
        }
    }
    if c.APIServer != nil {
        if err := c.APIServer.Shutdown(ctx); err != nil {
            logger.Error("API server shutdown error", "error", err)
        }
    }
    logger.Info("Shutdown completed")
}

1. Create `internal/lifecycle/service_windows.go` with `//go:build windows` tag (D-01):

```go
//go:build windows

package lifecycle

import (
    "context"
    "log/slog"
    "time"

    "golang.org/x/sys/windows/svc"
    "github.com/HQGroup/nanobot-auto-updater/internal/config"
)

// ServiceHandler implements svc.Handler for Windows service lifecycle (D-01).
type ServiceHandler struct {
    cfg        *config.Config
    logger     *slog.Logger
    version    string
    components *AppComponents
}

// NewServiceHandler creates a new service handler (D-01).
func NewServiceHandler(cfg *config.Config, logger *slog.Logger, version string) *ServiceHandler {
    return &ServiceHandler{
        cfg:     cfg,
        logger:  logger,
        version: version,
    }
}

// Execute is called by svc.Run to run the service (D-02, D-03).
// It reads ChangeRequests from r and reports Status to s.
func (h *ServiceHandler) Execute(args []string, r <-chan svc.ChangeRequest, s chan<- svc.Status) (svcSpecificEC bool, exitCode uint32) {
    const cmdsAccepted = svc.AcceptStop | svc.AcceptShutdown

    // D-02: Report starting state
    s <- svc.Status{State: svc.StartPending}

    // D-10: Initialize all business components
    components, err := AppStartup(h.cfg, h.logger, h.version)
    if err != nil {
        h.logger.Error("Service startup failed", "error", err)
        // Attempt cleanup of partial components
        cleanupCtx, cancel := context.WithTimeout(context.Background(), 30*time.Second)
        AppShutdown(cleanupCtx, components, h.logger)
        cancel()
        s <- svc.Status{State: svc.Stopped}
        return true, 1 // service-specific error code
    }
    h.components = components

    // D-02: Report running state, accept Stop and Shutdown
    s <- svc.Status{State: svc.Running, Accepts: cmdsAccepted}
    h.logger.Info("Service is running")

    // Main event loop -- wait for Stop/Shutdown (D-03)
loop:
    for {
        select {
        case c := <-r:
            switch c.Cmd {
            case svc.Interrogate:
                s <- c.CurrentStatus
            case svc.Stop, svc.Shutdown:
                h.logger.Info("Service stop requested", "cmd", c.Cmd)
                break loop
            default:
                // Ignore ParamChange, SessionChange, etc.
            }
        }
    }

    // D-02: Report stopping state
    s <- svc.Status{State: svc.StopPending}

    // D-04, D-06: Graceful shutdown with 30-second timeout
    shutdownCtx, cancel := context.WithTimeout(context.Background(), 30*time.Second)
    defer cancel()
    AppShutdown(shutdownCtx, components, h.logger)

    // D-02: Report stopped -- svc package handles process exit (D-12)
    s <- svc.Status{State: svc.Stopped}
    return false, 0
}
```

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| SCM -> ServiceHandler | Windows Service Control Manager sends control codes (Stop, Shutdown, Interrogate) via svc.ChangeRequest channel. These are trusted OS-level signals. |
| ServiceHandler -> AppComponents | Handler calls AppStartup/AppShutdown which manage all application components. Compromise of handler state could bypass shutdown. |
| Config -> ServiceHandler | ServiceHandler reads cfg.Service.ServiceName from config. Malformed config could cause unexpected service behavior. |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-47-01 | Tampering | Config -> ServiceHandler | mitigate | Config.Validate() already enforces alphanumeric-only ServiceName (Phase 46 D-10), max 256 chars. ServiceHandler does not modify config. |
| T-47-02 | Denial of Service | AppShutdown | mitigate | 30-second context timeout ensures shutdown completes even if a component hangs. APIServer.Shutdown(ctx) respects the context. Other Stop() calls are near-instant. |
| T-47-03 | Elevation of Privilege | ServiceHandler | accept | Service runs under the Windows account configured in Phase 48. Handler itself does not change privileges. |
| T-47-04 | Tampering | ServiceHandler.Execute | accept | Control codes come from SCM (trusted OS). No untrusted input reaches Execute. Non-handled codes (ParamChange etc.) are silently ignored. |
</threat_model>
