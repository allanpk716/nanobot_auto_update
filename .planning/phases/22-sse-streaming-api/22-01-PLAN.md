**3. handleSSE 函数 (核心逻辑):**
```go
func (h *SSEHandler) Handle(w http.ResponseWriter, r *http.Request) {
    // 1. 设置 SSE HTTP 头
    w.Header().Set("Content-Type", "text/event-stream")
    w.Header().Set("Cache-Control", "no-cache")
    w.Header().Set("Connection", "keep-alive")

    // 2. 检查 Flusher 支持
    flusher, ok := w.(http.Flusher)
    if !ok {
        http.Error(w, "Streaming not supported", http.StatusInternalServerError)
        return
    }

    // 3. 提取 instance 参数
    instanceName := r.PathValue("instance")
    if instanceName == "" {
        http.Error(w, "Instance name required", http.StatusBadRequest)
        return
    }

    // 4. 获取 LogBuffer
    logBuffer, err := h.instanceManager.GetLogBuffer(instanceName)
    if err != nil {
        http.Error(w, fmt.Sprintf("Instance %s not found", instanceName), http.StatusNotFound)
        return
    }

    // 5. 订阅日志流
    logChan := logBuffer.Subscribe()
    defer logBuffer.Unsubscribe(logChan)

    h.logger.Info("SSE client connected", "instance", instanceName)

    // 6. 发送 connected 事件
    fmt.Fprintf(w, "event: connected\ndata: {\"instance\":\"%s\"}\n\n", instanceName)
    flusher.Flush()

    // 7. 启动心跳 ticker (30秒)
    ticker := time.NewTicker(30 * time.Second)
    defer ticker.Stop()

    // 8. for-select 循环
    ctx := r.Context()
    for {
        select {
        case entry, ok := <-logChan:
            if !ok {
                // LogBuffer channel 已关闭
                h.logger.Info("LogBuffer channel closed", "instance", instanceName)
                return
            }
            // 发送日志事件
            h.writeSSEEvent(w, flusher, entry.Source, entry.Content)

        case <-ticker.C:
            // 发送心跳注释
            fmt.Fprint(w, ": ping\n\n")
            flusher.Flush()
            h.logger.Debug("SSE heartbeat sent", "instance", instanceName)

        case <-ctx.Done():
            // 客户端断开连接
            h.logger.Info("SSE client disconnected", "instance", instanceName)
            return
        }
    }
}
```

func (h *SSEHandler) writeSSEEvent(w http.ResponseWriter, flusher http.Flusher, eventType string, data string) {
    fmt.Fprintf(w, "event: %s\ndata: %s\n\n", eventType, data)
    flusher.Flush()
}

func TestSSEEndpoint(t *testing.T) {
    // 创建 mock InstanceManager
    im := instance.NewInstanceManager()
    lb := logbuffer.NewLogBuffer()
    im.SetLogBuffer("test", lb)

    // 创建 SSE handler
    handler := NewSSEHandler(im, logger.Default())

    // 创建测试请求
    req := httptest.NewRequest("GET", "/api/v1/logs/test/stream", nil)
    req.SetPathValue("instance", "test")
    rec := httptest.NewRecorder()

    // 启动 handler (在 goroutine 中)
    ctx, cancel := context.WithCancel(context.Background())
    req = req.WithContext(ctx)

    go func() {
        handler.Handle(rec, req)
    }()

    // 等待响应
    time.Sleep(100 * time.Millisecond)

    // 验证 HTTP 头
    if rec.Header().Get("Content-Type") != "text/event-stream" {
        t.Errorf("Expected Content-Type=text/event-stream, got %s", rec.Header().Get("Content-Type"))
    }
    if rec.Header().Get("Cache-Control") != "no-cache" {
        t.Errorf("Expected Cache-Control=no-cache, got %s", rec.Header().Get("Cache-Control"))
    }

    // 清理
    cancel()
}

func TestSSEEventFormat(t *testing.T) {
    im := instance.NewInstanceManager()
    lb := logbuffer.NewLogBuffer()
    im.SetLogBuffer("test", lb)

    handler := NewSSEHandler(im, logger.Default())

    req := httptest.NewRequest("GET", "/api/v1/logs/test/stream", nil)
    req.SetPathValue("instance", "test")
    rec := httptest.NewRecorder()

    ctx, cancel := context.WithCancel(context.Background())
    req = req.WithContext(ctx)

    go func() {
        handler.Handle(rec, req)
    }()

    time.Sleep(100 * time.Millisecond)

    // 写入测试日志
    lb.Write("stdout", "test stdout log")
    time.Sleep(50 * time.Millisecond)

    lb.Write("stderr", "test stderr log")
    time.Sleep(50 * time.Millisecond)

    cancel()
    time.Sleep(100 * time.Millisecond)

    body := rec.Body.String()

    // 验证事件格式
    if !strings.Contains(body, "event: connected") {
        t.Error("Expected connected event")
    }
    if !strings.Contains(body, "event: stdout") {
        t.Error("Expected stdout event")
    }
    if !strings.Contains(body, "test stdout log") {
        t.Error("Expected stdout log content")
    }
    if !strings.Contains(body, "event: stderr") {
        t.Error("Expected stderr event")
    }
    if !strings.Contains(body, "test stderr log") {
        t.Error("Expected stderr log content")
    }
}
