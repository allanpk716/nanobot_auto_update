```go
type Server struct {
    httpServer *http.Server
    logger     *slog.Logger
}

func NewServer(cfg *config.APIConfig, im *instance.Manager, logger *slog.Logger) (*Server, error) {
    if cfg == nil {
        return nil, fmt.Errorf("API config is nil")
    }
    if cfg.Port == 0 {
        return nil, fmt.Errorf("API port is required")
    }

    // 创建 SSE handler
    sseHandler := NewSSEHandler(im, logger)

    // 创建路由
    mux := http.NewServeMux()
    mux.HandleFunc("GET /api/v1/logs/{instance}/stream", sseHandler.Handle)

    // 创建 HTTP 服务器
    httpServer := &http.Server{
        Addr:         fmt.Sprintf(":%d", cfg.Port),
        Handler:      mux,
        WriteTimeout: 0, // SSE-07: 支持 SSE 长连接
        ReadTimeout:  10 * time.Second,
    }

    logger.Info("HTTP server created", "port", cfg.Port, "write_timeout", "0 (unlimited)")

    return &Server{
        httpServer: httpServer,
        logger:     logger,
    }, nil
}

func (s *Server) Start() error {
    s.logger.Info("HTTP server starting", "addr", s.httpServer.Addr)
    err := s.httpServer.ListenAndServe()
    if err != nil && err != http.ErrServerClosed {
        return err
    }
    return nil
}

func (s *Server) Shutdown(ctx context.Context) error {
    s.logger.Info("HTTP server shutting down")
    return s.httpServer.Shutdown(ctx)
}
```

func main() {
    // ... 现有代码: 加载配置,创建 logger,创建 InstanceManager ...

    // 创建 API 服务器 (SSE-07: WriteTimeout=0)
    if cfg.API != nil && cfg.API.Port != 0 {
        apiServer, err := api.NewServer(cfg.API, instanceManager, logger)
        if err != nil {
            logger.Error("Failed to create API server", "error", err)
            os.Exit(1)
        }

        // 启动 API 服务器 (goroutine)
        go func() {
            logger.Info("Starting API server", "port", cfg.API.Port)
            if err := apiServer.Start(); err != nil {
                logger.Error("API server error", "error", err)
            }
        }()
    }

    // 设置优雅关闭
    sigChan := make(chan os.Signal, 1)
    signal.Notify(sigChan, syscall.SIGINT, syscall.SIGTERM)

    // 等待关闭信号
    <-sigChan
    logger.Info("Shutdown signal received")

    // 优雅关闭 (with timeout)
    shutdownCtx, cancel := context.WithTimeout(context.Background(), 10*time.Second)
    defer cancel()

    // 停止 scheduler
    if sched != nil {
        sched.Stop()
    }

    // 关闭 API 服务器
    if apiServer != nil {
        if err := apiServer.Shutdown(shutdownCtx); err != nil {
            logger.Error("API server shutdown error", "error", err)
        }
    }

    logger.Info("Shutdown completed")
}
