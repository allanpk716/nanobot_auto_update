   func (ic *InstanceConfig) ShouldAutoStart() bool {
       // 由于 viper 会将未指定的 bool 解析为零值 false
       // 我们无法区分 "未指定" 和 "显式 false"
       // 解决方案：使用指针类型 *bool 或接受 viper 的行为

       // 正确理解：viper 未指定时为零值 false，但我们希望默认启动
       // 解决方案：在 config.go Load() 中无法区分，需要使用 *bool 或改变语义
       //
       // 最终方案：根据 CONTEXT.md 用户决策，"auto_start: false 的实例跳过"
       // 默认行为：所有实例都启动，只有显式 auto_start: false 才跳过
