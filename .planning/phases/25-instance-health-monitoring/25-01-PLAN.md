       // InstanceHealthState tracks instance health
       type InstanceHealthState struct {
           IsRunning bool
           LastCheck time.Time
       }

       // HealthMonitor manages health check loop
       type HealthMonitor struct {
           instances []config.InstanceConfig
           interval  time.Duration
           logger    *slog.Logger
           states    map[string]*InstanceHealthState
           mu        sync.RWMutex
           ctx       context.Context
           cancel    context.CancelFunc
       }

       // NewHealthMonitor creates a new health monitor
       func NewHealthMonitor(
           instances []config.InstanceConfig,
           interval time.Duration,
           logger *slog.Logger,
       ) *HealthMonitor

       // Start begins the health check loop (runs in goroutine)
       func (hm *HealthMonitor) Start()
           - Uses time.Ticker with defer ticker.Stop()
           - Immediate first check, then tick-based checks
           - select with ctx.Done() for graceful shutdown
           - Chinese logs: "健康监控已启动", "健康监控已停止"

       // checkAllInstances iterates all instances
       func (hm *HealthMonitor) checkAllInstances()

       // checkInstance checks single instance and logs state changes
       func (hm *HealthMonitor) checkInstance(inst config.InstanceConfig)
           - Calls lifecycle.IsNanobotRunning(inst.Port)
           - On first check: records state, logs "初始状态检查" INFO
           - On state change running->stopped: logs "实例已停止" ERROR
           - On state change stopped->running: logs "实例已恢复运行" INFO
           - Uses sync.RWMutex for concurrent state map access
           - Only logs state change when state actually changes (not on every check)

       // Stop gracefully stops the monitor
       func (hm *HealthMonitor) Stop()
