     ```go
     func NewLogBuffer(logger *slog.Logger) *LogBuffer {
         return &LogBuffer{
             entries:     [5000]LogEntry{},
             subscribers: make(map[chan LogEntry]context.CancelFunc),  // 新增
             logger:      logger.With("component", "logbuffer"),
         }
     }
     ```
   - 扩展 Write 方法,添加非阻塞发送到订阅者:
     ```go
     func (lb *LogBuffer) Write(entry LogEntry) error {
         lb.mu.Lock()
         defer lb.mu.Unlock()

         // 原有环形缓冲区写入逻辑 (Plan 01 实现)
         lb.entries[lb.head] = entry
         lb.head = (lb.head + 1) % 5000
         if lb.size < 5000 {
             lb.size++
         }

         // 新增: 非阻塞发送到所有订阅者 (CONTEXT.md 约束)
         for ch, cancel := range lb.subscribers {
             select {
             case ch <- entry:
                 // 发送成功
             default:
                 // Channel 满,丢弃该订阅者的这条日志 (不阻塞)
                 lb.logger.Warn("Subscriber channel full, dropping log",
                     "channel_capacity", cap(ch))
             }
         }

         return nil
     }
     ```

2. 创建 `internal/logbuffer/subscriber.go`:
   - 实现 Subscribe 方法 (返回只读 channel):
     ```go
     func (lb *LogBuffer) Subscribe() <-chan LogEntry {
         lb.mu.Lock()
         defer lb.mu.Unlock()

         // 创建订阅者 channel (容量 100,CONTEXT.md 约束)
         ch := make(chan LogEntry, 100)

         // 创建可取消的 context
         ctx, cancel := context.WithCancel(context.Background())
         lb.subscribers[ch] = cancel

         // 启动订阅者 goroutine
         go lb.subscriberLoop(ctx, ch)

         return ch
     }
     ```
   - 实现 subscriberLoop 方法 (先发历史,再等实时):
     ```go
     func (lb *LogBuffer) subscriberLoop(ctx context.Context, ch chan<- LogEntry) {
         defer close(ch) // goroutine 退出时关闭 channel

         // 1. 先发送历史日志 (CONTEXT.md 约束)
         history := lb.GetHistory()
         for _, entry := range history {
             select {
             case ch <- entry:
                 // 发送成功
             case <-ctx.Done():
                 // Unsubscribe 被调用,停止发送
                 return
             }
         }

         // 2. 实时日志由 Write() 方法直接发送到 ch
         // 这里只需要等待 context 取消
         <-ctx.Done()
     }
     ```
   - 实现 Unsubscribe 方法 (接收 channel 作为句柄):
     ```go
     func (lb *LogBuffer) Unsubscribe(ch <-chan LogEntry) {
         lb.mu.Lock()
         defer lb.mu.Unlock()

         // 类型断言: 只读 channel 转换为读写 channel
         chWrite := chan LogEntry(ch)

         if cancel, exists := lb.subscribers[chWrite]; exists {
             cancel()                      // 取消 context,停止 goroutine
             delete(lb.subscribers, chWrite) // 从 map 中删除
         }
     }
     ```

**关键约束:**
- 订阅者 channel 容量必须为 100 (CONTEXT.md 约束)
- Write 发送日志必须使用 `select + default` 非阻塞 (CONTEXT.md 约束)
- 新订阅者必须先接收历史日志 (CONTEXT.md 约束)
- Unsubscribe 必须关闭 channel 并停止 goroutine (防止泄漏)
- 丢弃慢订阅者日志时记录 WARN (Claude's Discretion,推荐记录)

**避免:**
- 不要使用无缓冲 channel (会阻塞 Write)
- 不要忘记 `defer close(ch)` (goroutine 泄漏)
- 不要在 subscriberLoop 中直接读取实时日志 (由 Write 发送)
- 不要在 Unsubscribe 中关闭 channel (由 subscriberLoop 的 defer close 关闭)
